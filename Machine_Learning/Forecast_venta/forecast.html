from statsmodels.tsa.arima.model import ARIMA
import numpy as np

# Definir una funcin para entrenar modelos con diferentes combinaciones de p, d, q
def ajustar_arima(data, p_values, d_values, q_values):
    best_aic = np.inf
    best_model = None
    best_order = None

    for p in p_values:
        for d in d_values:
            for q in q_values:
                try:
                    modelo = ARIMA(data, order=(p, d, q))
                    modelo_fit = modelo.fit()
                    aic = modelo_fit.aic
                    if aic < best_aic:
                        best_aic = aic
                        best_model = modelo_fit
                        best_order = (p, d, q)
                except:
                    continue

    return best_model, best_order, best_aic

# Probar diferentes valores de p, d, q
p_values = [0, 1, 2, 3, 4, 5]
d_values = [0, 1]
q_values = [0, 1, 2]

best_model, best_order, best_aic = ajustar_arima(ventas_diarias['Vta $'], p_values, d_values, q_values)

# Mostrar el mejor modelo y el orden elegido
print(f"Mejor modelo ARIMA: {best_order}")
print(f"AIC del mejor modelo: {best_aic}")
